Introduction

This report presents a descriptive analysis of data collected from a survey on streaming habits. Participants answered questions about their viewing behavior, mood, and platform preferences.

We use basic statistics and visualizations to explore relationships between variables and identify patterns.

1. Enjoyability & Game Duration

ggplot(survey_data, aes(x = as.factor(on_a_scale_of_1_5_how_enjoyable_was_the_game))) +
  geom_bar(fill = "darkorchid") +
  labs(title = "Game Enjoyability (1–5 Scale)", x = "Enjoyment Rating", y = "Count")
# Count ratings
enjoy_counts <- survey_data %>%
  count(on_a_scale_of_1_5_how_enjoyable_was_the_game) %>%
  mutate(percentage = n / sum(n),
         label = paste0(on_a_scale_of_1_5_how_enjoyable_was_the_game, " (", round(percentage * 100), "%)"))

# donut chart
ggplot(enjoy_counts, aes(x = 2, y = percentage, fill = as.factor(on_a_scale_of_1_5_how_enjoyable_was_the_game))) +
  geom_col(width = 1, color = "white") +
  coord_polar(theta = "y") +
  xlim(0.5, 2.5) +  # Creates the donut hole
  theme_void() +
  geom_text(aes(label = label), position = position_stack(vjust = 0.5), size = 4) +
  scale_fill_brewer(palette = "Dark2") +
  labs(title = "Game Enjoyability (Donut Chart)", fill = "Rating")
survey_data %>%
  count(how_long_did_you_play_the_game) %>%
  ggplot(aes(x = reorder(how_long_did_you_play_the_game, -n), y = n)) +
  geom_col(fill = "darkcyan") +
  labs(title = "How Long Did Players Play the Game?", x = "Play Time", y = "Count")
# boxplot with outlier detection for play time categories
ggplot(survey_data, aes(x = how_long_did_you_play_the_game, 
                        y = as.numeric(on_a_scale_of_1_5_how_enjoyable_was_the_game))) +
  geom_boxplot(fill = "darkcyan", outlier.color = "red", outlier.shape = 8) +
  labs(title = "Enjoyability by Play Time (Boxplot)", 
       x = "Play Time Duration", 
       y = "Enjoyment Rating (1–5)") +
  theme_minimal()
